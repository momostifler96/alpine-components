Intégration Livewire

Tous les composants dispatche les événements input et change pour une compatibilité native avec wire:model.

{{-- Select multiple dans un composant Livewire --}}
<div
    x-data="apSelect(@js(['options' => $options, 'multiple' => true]))"
    x-modelable="value"
    wire:model.live="selectedTechs"
>
    {{-- Trigger --}}
    <button @click="toggleOpen()">...</button>

    {{-- Options --}}
    <div x-show="open">...</div>
</div>

{{-- Switch --}}
<div x-data="apSwitch()" x-modelable="value" wire:model.live="darkMode">
    <button @click="toggle()">...</button>
</div>

{{-- Slider --}}
<div x-data="apSlider({ min: 0, max: 100 })" x-modelable="value" wire:model.live="volume">
    <input type="range" x-model.number="value" @input="onInput($event)">
</div>
